import os
import shutil

class CopiadorDeArquivos:
    def __init__(self, caminho_origem, caminho_destino, prefixo):
        self.caminho_origem = caminho_origem
        self.caminho_destino = caminho_destino
        self.prefixo = prefixo

    def copiar_arquivos(self):
        for arquivo in os.listdir(self.caminho_origem):
            if arquivo.startswith(self.prefixo):
                origem_completa = os.path.join(self.caminho_origem, arquivo)
                destino_completa = os.path.join(self.caminho_destino, arquivo)
                
                if not os.path.exists(destino_completa):
                    self.copiar(origem_completa, self.caminho_destino)
                    print(f'Arquivo {arquivo} copiado com sucesso!')
                else:
                    print(f'Arquivo {arquivo} j√° existe no destino.')

    def copiar(self, origem, destino):
        shutil.copy(origem, destino)

def main():
    caminho_origem = '/caminho/origem'
    caminho_destino = '/caminho/destino'
    prefixo = '11094230'
    
    copiador = CopiadorDeArquivos(caminho_origem, caminho_destino, prefixo)
    copiador.copiar_arquivos()

if __name__ == '__main__':
    main()
