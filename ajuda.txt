import os
import shutil

# Caminho de origem e destino
caminho_origem = '/caminho/origem'
caminho_destino = '/caminho/destino'

# Verifique todos os arquivos no diretório de origem
for arquivo in os.listdir(caminho_origem):
    # Verifique se o arquivo tem o prefixo desejado
    if arquivo.startswith('11094230'):
        # Construa os caminhos completos de origem e destino
        origem_completa = os.path.join(caminho_origem, arquivo)
        destino_completa = os.path.join(caminho_destino, arquivo)
        
        # Verifique se o arquivo já existe no diretório de destino
        if not os.path.exists(destino_completa):
            # Copie o arquivo para o diretório de destino
            shutil.copy(origem_completa, caminho_destino)
            print(f'Arquivo {arquivo} copiado com sucesso!')
        else:
            print(f'Arquivo {arquivo} já existe no destino.')
